# Start MySql Client with User root and Passwort prompt

`mysql --user=root -p`

# Show existing databases

`show databases;`

# Switch context to database

`use mysql;`

# Show tables

`show tables;`

# Show information of table 'user'

`describe user;`

# Select Host and User data from table 'user'

`select host, user from user;`

# Exit

`exit;`
